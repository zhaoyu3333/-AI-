    <script>
        // 切换功能模块
        function switchModule(id) {
            // 隐藏所有模块
            let modules = document.querySelectorAll('.module');
            modules.forEach(item => {
                item.classList.remove('show');
            });
            // 取消按钮激活状态
            let btns = document.querySelectorAll('.nav-btn');
            btns.forEach(item => {
                item.classList.remove('active');
            });
            // 显示选中模块
            document.getElementById(id).classList.add('show');
            event.target.classList.add('active');
        }

        // AI情绪分析模拟
        function analysisEmo() {
            let input = document.getElementById('emoInput').value.trim();
            let resBox = document.getElementById('emoResult');
            if (input === "") {
                resBox.innerText = "请先输入你的心情和感受哦！";
                return;
            }
            // 简单关键词匹配模拟AI识别
            if (input.includes("难过") || input.includes("低落") || input.includes("不开心")) {
                resBox.innerText = "AI识别结果：当前情绪偏向低落、伤感。建议多和朋友倾诉，也可以试试呼吸放松训练，调整状态~";
            } else if (input.includes("焦虑") || input.includes("压力") || input.includes("烦躁")) {
                resBox.innerText = "AI识别结果：当前存在焦虑、压力过大的状态。适度放松，合理拆分目标，不要给自己过多负担。";
            } else if (input.includes("开心") || input.includes("快乐") || input.includes("顺利")) {
                resBox.innerText = "AI识别结果：情绪积极愉悦，状态良好！继续保持乐观的心态哦。";
            } else {
                resBox.innerText = "AI识别结果：情绪状态平稳。日常多关注自身感受，学会调节情绪，保持身心健康。";
            }
        }

        // 心理问答模拟回复
        function chatReply() {
            let input = document.getElementById('chatInput').value.trim();
            let resBox = document.getElementById('chatResult');
            if (input === "") {
                resBox.innerText = "请输入你的困惑，我来为你解答~";
                return;
            }
            resBox.innerText = "AI助手：收到你的问题啦。遇到烦恼是成长的常态，试着换个角度看待问题，也可以多和身边人沟通。如果困扰难以排解，记得联系心理老师或拨打12355求助热线。";
        }

        // 放松训练引导
        function relaxTip() {
            let resBox = document.getElementById('relaxResult');
            resBox.innerText = "放松引导：请找安静的环境坐好，双肩下沉，闭上眼睛。跟随节奏：吸气4秒 → 屏住呼吸4秒 → 缓缓呼气6秒，重复5组，慢慢平复身心。";
        }
